find_multi_script_group <- function  (supply_date, ndays ) {
  overlap = 1
  pos = 1
  group_id = 0
  rv=c()
  while (pos <= length( ndays )) {
    start_group = pos
    end_date = supply_date[ pos ] + ndays[ pos ] + overlap

    pos = pos + 1
    in_session=TRUE
    while ( (pos <= length( ndays )) &
           (supply_date[pos] - overlap < end_date) 
         ) {
      end_date = supply_date[ pos ] + ndays[ pos ] + overlap
      pos = pos + 1
    }
    rv = c(rv, rep( group_id, pos - start_group  ))
    group_id = group_id + 1
  }
  rv
}
